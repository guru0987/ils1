Self-Supervised Learning (SSL) Code

import torch
import torch.nn as nn
import torch.nn.functional as F
import torchvision.transforms as transforms
from torchvision.datasets import CIFAR10
from torch.utils.data import DataLoader
from sklearn.manifold import TSNE
import matplotlib.pyplot as plt

class SimpleAugmentation:
    def __init__(self):
        self.transform = transforms.Compose([
            transforms.RandomHorizontalFlip(),
            transforms.ToTensor()
        ])

    def __call__(self, x):
        return self.transform(x), self.transform(x)

dataset = CIFAR10(
    root="./data",
    train=True,
    download=True,
    transform=SimpleAugmentation()
)

loader = DataLoader(dataset, batch_size=32, shuffle=True)

class SimpleEncoder(nn.Module):
    def __init__(self):
        super(SimpleEncoder, self).__init__()
        self.net = nn.Sequential(
            nn.Conv2d(3, 16, 3, stride=2, padding=1),
            nn.ReLU(),
            nn.Conv2d(16, 32, 3, stride=2, padding=1),
            nn.ReLU(),
            nn.AdaptiveAvgPool2d((1, 1)),
            nn.Flatten()
        )

    def forward(self, x):
        return self.net(x)

class SSLModel(nn.Module):
    def __init__(self):
        super(SSLModel, self).__init__()
        self.encoder = SimpleEncoder()
        self.projector = nn.Linear(32, 16)

    def forward(self, x):
        features = self.encoder(x)
        return self.projector(features)

def contrastive_loss(z1, z2):
    z1 = F.normalize(z1, dim=1)
    z2 = F.normalize(z2, dim=1)
    return 1 - F.cosine_similarity(z1, z2).mean()

device = "cuda" if torch.cuda.is_available() else "cpu"
print("Using device:", device)

model = SSLModel().to(device)
optimizer = torch.optim.Adam(model.parameters(), lr=0.001)

epochs = 2

for epoch in range(epochs):
    total_loss = 0.0
    for (x1, x2), _ in loader:
        x1, x2 = x1.to(device), x2.to(device)

        z1 = model(x1)
        z2 = model(x2)

        loss = contrastive_loss(z1, z2)

        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

        total_loss += loss.item()

    print(f"Epoch {epoch+1}/{epochs}, Loss: {total_loss/len(loader):.4f}")

model.eval()

with torch.no_grad():
    batch = next(iter(loader))
    (x1, x2), labels = batch

    imgs = x1.to(device)
    features = model.encoder(imgs).cpu().numpy()
    labels = labels.numpy()

print("Feature Shape:", features.shape)

tsne = TSNE(n_components=2, random_state=42)
features_2d = tsne.fit_transform(features)

plt.figure(figsize=(6, 6))

for i in range(10):
    idx = labels == i
    plt.scatter(features_2d[idx, 0], features_2d[idx, 1], label=str(i), alpha=0.6)

plt.legend()
plt.title("SSL Feature Embeddings (t-SNE)")
plt.xlabel("t-SNE Dim 1")
plt.ylabel("t-SNE Dim 2")
plt.show()
